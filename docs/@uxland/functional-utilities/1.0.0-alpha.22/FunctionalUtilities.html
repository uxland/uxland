Constant Builder

Example
constantBuilder('PREFIX')('CONSTANT') //=> 'PREFIX:CONSTANT'
constantBuilder('PREFIX', 'SUFFIX')('CONSTANT') //=> 'PREFIX:CONSTANT:SUFFIX'
constantBuilder('PREFIX', 'SUFFIX', '$$')('CONSTANT') //=> 'PREFIX$$CONSTANT$$SUFFIX'
