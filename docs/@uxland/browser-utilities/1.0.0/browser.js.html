/**
 * Returns if it's an android device
 * @function
 * @name isAndroid
 * @memberof BrowserUtilities
 * @since v1.0.0
 * @returns {boolean}
 * @example
 *
 * isAndroid(); //=> true | false
 */
const isAndroid = () => {
    const ua = navigator.userAgent.toLowerCase();
    return ua.indexOf('android') > -1; //&& ua.indexOf("mobile");
};
exports.isAndroid = isAndroid;
