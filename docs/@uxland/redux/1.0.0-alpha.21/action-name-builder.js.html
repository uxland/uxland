const functional_utilities_1 = require("@uxland/functional-utilities");
/**
 * Action name builder
 * @function
 * @memberof Redux
 * @name actionNameBuilder
 * @since v1.0.0
 * @param {string} prefix Action prefix
 * @param {string=} separator String separator between prefix, constant and suffix
 * @returns {FunctionalUtilities.ConstantBuilder}
 * @see FunctionalUtilities.ConstantBuilder
 * @example
 *
 * actionNameBuilder('PREFIX', '@@')('NAME') //=> 'PREFIX@@NAME@@ACTION'
 *
 */
exports.actionNameBuilder = (prefix, separator) => {
    const builder = functional_utilities_1.constantBuilder(prefix, 'ACTION', separator);
    return (name) => builder(name);
};
